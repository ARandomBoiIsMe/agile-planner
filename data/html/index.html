include: __CURR_CONFIG__, __LOG__, __HTML__

# Outputs all binary codes of a specified length
str: ""
func binary(bin, x)
  if(x.==(0))
    str.concat(bin, "\n")
    return
  x.--()
  binary(bin.add("0"), x)
  binary(bin.add("1"), x)

print("Enter number: ")
x: input_int()
binary("", x)

write_file("data/bin.txt", str)
